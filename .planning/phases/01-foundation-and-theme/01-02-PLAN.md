**2. `src/theme/ThemeProvider.tsx`** -- React Context provider and useTheme hook.

```typescript
import React, { createContext, useContext, useMemo } from 'react';
import { darkTheme, type Theme } from './tokens';

const ThemeContext = createContext<Theme>(darkTheme);

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  // Currently dark-only; wrapping in useMemo for future light mode support
  const theme = useMemo(() => darkTheme, []);
  return (
    <ThemeContext.Provider value={theme}>
      {children}
    </ThemeContext.Provider>
  );
}

export function useTheme(): Theme {
  return useContext(ThemeContext);
}
```

IMPORTANT: The context default value is `darkTheme` (not `null` or `undefined`). This means `useTheme()` works even outside ThemeProvider (for testing), but ThemeProvider is still required in the app tree for proper usage.

```typescript
import 'expo-sqlite/localStorage/install';
import { createClient } from '@supabase/supabase-js';

const supabaseUrl = process.env.EXPO_PUBLIC_SUPABASE_URL!;
const supabaseAnonKey = process.env.EXPO_PUBLIC_SUPABASE_ANON_KEY!;

if (!supabaseUrl || !supabaseAnonKey) {
  throw new Error(
    'Missing Supabase env vars. Create .env with EXPO_PUBLIC_SUPABASE_URL and EXPO_PUBLIC_SUPABASE_ANON_KEY'
  );
}

export const supabase = createClient(supabaseUrl, supabaseAnonKey, {
  auth: {
    storage: localStorage,
    autoRefreshToken: true,
    persistSession: true,
    detectSessionInUrl: false, // Required for React Native -- no URL-based auth
  },
});
```

CRITICAL requirements:
- `import 'expo-sqlite/localStorage/install'` MUST be the first import (side-effect that polyfills `localStorage`)
- Use `process.env.EXPO_PUBLIC_SUPABASE_URL` with DOT notation, NOT bracket notation (Expo performs static text replacement at build time)
- Set `detectSessionInUrl: false` -- React Native has no URL hash for auth tokens
- Set `storage: localStorage` -- uses the expo-sqlite polyfill installed above
- Auth token storage uses expo-sqlite/localStorage per user decision (NOT expo-secure-store which has 2048-byte limit)
- Runtime validation throws if env vars are missing (catches misconfigured .env)
